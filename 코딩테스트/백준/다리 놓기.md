import sys

T = int(sys.stdin.readline())
for i in range(T):
    N, M = list(map(int,sys.stdin.readline().split()))
    den = 1 # 분모(N!)
    num = 1 # 분자(M*M-1*...M-N)
    for j in range(1,N+1):
        den*=j # 팩토리얼
    for j in range(N):
        num*=(M-j) # M*M-1*...M-N
    #print(den, num)
    print(num//den)
